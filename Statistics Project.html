# Step 5: Formulate Hypotheses

from scipy import stats

# Do males and females spend differently?

# H0(Null Hypothesis): There is no significant difference in average MonthlySpend between males and females
# Ha(Alternate Hypothesis): There is a significant difference in average MonthlySpend between males and females

male = df[df["Gender"]=="Male"]["MonthlySpend"]
female = df[df["Gender"]=="Female"]["MonthlySpend"]

t_stat, p_value = stats.ttest_ind(male,female)

print("1. Gender vs MonthlySpend (T-Test)")
print("T-Statistic:", t_stat)
print("P-Value:", p_value)

if p_value < 0.05:
    print("Reject the H0 : Significant difference in average Monthly Spend b/w Males and Females")
else:
    print("Accept the H0 : No significant differrent found b/w Males and Females average Monthly Spend")

# 2. Does education level impact average monthly spend? 

# H0(Null Hypothesis): Average MonthlySpend is same across all education levels
# Ha(Alternate Hypothesis): Average MonthlySpend is different across all education levels

edu_group = [df[df["Education"]==e]["MonthlySpend"] for e in df["Education"].unique()]

f_stat, p_value = stats.f_oneway(*edu_group)

print("2. Education vs MonthlySpend (ANOVA)")
print("F-Statistic:", f_stat)
print("P-Value:", p_value)

if p_value < 0.05:
